Самый простой скрипт сбора почты по POP3:

{% highlight Python %} #!/usr/bin/env python3 import poplib def prompt(prompt): return input(prompt).strip() server = prompt("POP3 server: ") user = prompt("POP3 user: ") password = prompt("POP3 pass: ") pp = poplib.POP3(server) pp.user(user) pp.pass_(password) numMessages = len(pp.list()[1]) for i in range(numMessages): for j in pp.retr(i+1)[1]: print(j) {% endhighlight %}
